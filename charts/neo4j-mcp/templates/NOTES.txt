Neo4j MCP server has been deployed.

Get the application URL by running:
{{- if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward svc/{{ include "neo4j-mcp.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc {{ include "neo4j-mcp.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if eq .Values.service.type "LoadBalancer" }}
  export SERVICE_IP=$(kubectl get svc {{ include "neo4j-mcp.fullname" . }} -o jsonpath="{.status.loadBalancer.ingress[0].ip}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- end }}
